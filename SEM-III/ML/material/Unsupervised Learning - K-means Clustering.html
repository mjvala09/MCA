# Load multivar data in the input file
def load_data(input_file):
    x = []
    y = []
    with open(input_file, 'r') as f:
        for line in f.readlines():
            data = [float(i) for i in line.split(',')]
            x.append(data[0])
            y.append(data[1])
            
    x=np.array(x)
    y=np.array(y)
    data=list(zip(x,y))
    data=[list(i) for i in data]
    return data

data=load_data("usldata.txt")
